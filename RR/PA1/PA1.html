nNZimpute <- 0
for (i in 1:ns) {
    if (A$naSteps[i]) {
        nI <- nI + 1
        for (j in 1:length(medianStepsByInterval)) {
            k <- as.integer(names(medianStepsByInterval[j]))
            m <- medianStepsByInterval[j]
            if (A$interval[i] == k) {
                if (m == 0) {
                  nZimpute <- nZimpute + 1
                } else {
                  nNZimpute <- nNZimpute + 1
                }
                A$impSteps <- m
                break
            }
        }
